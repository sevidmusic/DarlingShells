general {
        colors = true
        interval = 1
        color_good = "#00e64d"
        color_bad = "#b30059"
        color_degraded = "#c61aff"
}

order += "load"
order += "ethernet _first_"
order += "wireless _first_"
order += "tztime local"
# order += "ipv6"
# order += "disk /"
# order += "battery all"

wireless _first_ {
        # format_up = "W: (%quality at %essid) %ip"
        format_up = "Wifi: %essid %quality | ip:  %ip"
        format_down = "W: down"
}

ethernet _first_ {
        # if you use %speed, i3status requires root privileges
        format_up = "Ether: %ip (%speed)"
        # set empty string to hide when ethernet is down or not in use
        format_down = ""
}

battery all {
        format = "%status %percentage %remaining"
}

tztime local {
        format = "%m-%d-%Y %I:%M:%S %p %Z"
}

load {
        format = "%1min"
}

disk "/" {
        format = "%avail"
}
